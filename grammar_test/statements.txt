======
for statement
======

for (;;) {}
for ;; {}

---

(program
	(for_statement)
	(for_statement))

======
for in statement
======

for x in y {}
for x: Int in y {}

---

(program
	(for_in_statement (identifier) (identifier))
	(for_in_statement (identifier) (type (identifier)) (identifier)))


======
switch statement
======

switch x {
case _:
default:
}

---

(program
	(switch_statement (identifier)
		(case_statement (wildcard_pattern))
		(case_statement)))
